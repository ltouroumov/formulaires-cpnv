	\left\{
		\begin{array}{rcl}
			\formula{x_x(t)}{x_0+v_0\,\cos{\alpha}\,t} \\

		\end{array}

	\end{equation*}

\nextcol
	
	Port\'ee
	\begin{equation*}
		\left\{
		\begin{array}{ll}
			p=\frac{v_0^2\,\sin{2\alpha}}{a} & \text{Si}\:y_0 = 0 \\
			p=\text{solve}\left(\left\{
					\begin{array}{rcl}
						y(t) & = & 0 \\

				\text{en} \: n
			\right)

		\end{array}
		\right.
	\end{equation*}
